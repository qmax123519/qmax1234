    window.addEventListener('scroll', () => {

        const screen = window.innerHeight * 0.9;
        if (pos < screen) {
          el.classList.add('visible');
        }
      });
    });

      document.getElementById('panel').classList.add('active');

      document.getElementById('panel').classList.remove('active');
